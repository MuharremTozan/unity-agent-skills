using UnityEngine;

namespace Standard.Patterns.Singleton {
    /// <summary>
    /// StaticInstance: A simple global reference that doesn't enforce uniqueness or cleanup.
    /// Useful for systems where you know only one will exist but don't want the overhead.
    /// </summary>
    public abstract class StaticInstance<T> : MonoBehaviour where T : MonoBehaviour {
        public static T Instance { get; private set; }
        protected virtual void Awake() => Instance = this as T;

        protected virtual void OnApplicationQuit() {
            Instance = null;
            Destroy(gameObject);
        }
    }

    /// <summary>
    /// Singleton: Enforces a single instance by destroying any NEW instances that try to spawn.
    /// </summary>
    public abstract class Singleton<T> : StaticInstance<T> where T : MonoBehaviour {
        protected override void Awake() {
            if (Instance != null) {
                Debug.Log($"[Singleton] Fail: Duplicate instance of {typeof(T).Name} destroyed on {gameObject.name}");
                Destroy(gameObject);
                return;
            }
            base.Awake();
            Debug.Log($"[Singleton] Status: {typeof(T).Name} initialized | Data: Instance set");
        }
    }

    /// <summary>
    /// PersistentSingleton: A Singleton that persists across scene loads.
    /// </summary>
    public abstract class PersistentSingleton<T> : Singleton<T> where T : MonoBehaviour {
        protected override void Awake() {
            base.Awake();
            if (Instance == this) {
                transform.SetParent(null); // Ensure it's at root for DDOL
                DontDestroyOnLoad(gameObject);
            }
        }
    }
}
